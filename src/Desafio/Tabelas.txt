
CREATE DATABASE POSTO.FDB
 
CREATE TABLE COMBUSTIVEL (
	ID INTEGER,
	TIPO_COMBUSTIVEL VARCHAR(100),
        TANQUE VARCHAR(100),
	VALOR NUMERIC(15,2),
	CONSTRAINT COMBUSTIVEL_PK PRIMARY KEY (ID)
);

CREATE TABLE BOMBA (
	ID INTEGER,
	ID_COMBUSTIVEL INTEGER,
	CONSTRAINT BOMBA_PK PRIMARY KEY (ID),
	CONSTRAINT BOMBA_FK FOREIGN KEY (ID_COMBUSTIVEL) REFERENCES COMBUSTIVEL(ID)
);


CREATE TABLE ABASTECIMENTO (
	ID INTEGER GENERATED BY DEFAULT AS IDENTITY,
	BOMBA NUMERIC,
        TANQUE VARCHAR(100),
	VALOR_TOTAL NUMERIC(15,2),
	IMPOSTO NUMERIC(15,2),
	VALOR NUMERIC(15,2),
        LITROS NUMERIC(15,2),
	TIPO_COMBUSTIVEL VARCHAR(100),
	"DATA" DATE
);
